步骤 命令
第一步 /hd01/ronds/cassandra/bin/cqlsh
第二步 CREATE KEYSPACE eimp WITH replication={‘class’:’SimpleStrategy’, ‘replication_factor’:3};
第三步 USE eimp;
第四步 [!1]创建时标表
CREATE table if not exists eimp.timescale (
id uuid,
datatype int,
Value blob,
timebase bigint,
time timestamp ,
properties map<text, text>,
PRIMARY KEY (id,datatype)
)
with clustering order by (datatype desc)
AND compaction = {‘compaction_window_size’: ‘7’,
‘compaction_window_unit’: ‘DAYS’,
‘class’: ‘org.apache.cassandra.db.compaction.TimeWindowCompactionStrategy’};
第五步 [!1]创建工艺数据表
CREATE table if not exists eimp.processdata (
id uuid,
value float,
time timestamp ,
properties map<text, text>,
PRIMARY KEY (id,time)
)
with clustering order by (time desc)
AND compaction = {‘compaction_window_size’: ‘7’,
‘compaction_window_unit’: ‘DAYS’,
‘class’: ‘org.apache.cassandra.db.compaction.TimeWindowCompactionStrategy’};
第六步 [!1]创建特征数据
CREATE table if not exists eimp.traitdata (
id uuid,
value float,
time timestamp ,
properties map<text, text>,

PRIMARY KEY ((id,datatype),time)

with clustering order by (time desc)
AND compaction = {‘compaction_window_size’: ‘7’,
‘compaction_window_unit’: ‘DAYS’,
‘class’: ‘org.apache.cassandra.db.compaction.TimeWindowCompactionStrategy’};
