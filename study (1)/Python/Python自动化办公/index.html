字符串

"""
多行注释
"""
print("你好,世界")
print("你好,世界",end="\n")                # print默认end="\n"换行
print("你好,世界",end="\t")                # end="\t" tab-四个空格
print("你好","世界",sep=" ")               # print默认sep=" "空格
print("你好","世界",sep="---")             # 设置---连接
print("你好"+"3")                          # 在字符串中+代表连接
print("你好"*3)                            # 在字符串中*代表重复
print("I'LOVE'YOU")                       # "" '' 需成对出现
print('I"LOVE"YOU')                       # "" '' 需成对出现
print(r"\n你好,世界",end="\n")             # 取消转义

变量

"""
变量名规则：支持汉字，并且也可以由字母、数字和下划线构成，
但不能以数字开头，且区分大小写。
"""

a = 1
b = 2
print(b)
print(a)

c=d=f = 3     # 同时给多个变量赋值
print(c)
print(d)
print(f)

x = 4
y = 5
x,y = y,x     # 交换变量赋值
print(x)
print(y)

姓名 = "吴志明"
年龄 = 26
print(f'我的名字叫{姓名},今年{年龄}岁')    #f 直接引用变量
print(f'我的名字叫{姓名},明年{年龄+1}岁')  #变量可以计算

input输入

"""
input需要转换才能运算
"""

姓名 = input("请输入你的姓名:")
年龄 = int(input("请输入你的年龄:"))
print(f'{姓名},{年龄+1}')
